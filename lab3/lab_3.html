### APPROXIMATE GRADIENT
def df1(x,h1):
    return (f([x[0] + h1, x[1]]) - f([x[0] - h1, x[1]])) / (2 * h1)

def df2(x,h2):
    return (f([x[0], x[1] + h2]) - f([x[0], x[1] - h2])) / (2 * h2)


def df(x,h):
    return np.array([df1(x,h[0]), df2(x,h[1])])

### TRUE GRADIENT (ONLY FOR QUADRATIC FUNCTIONS)
def truedf(x):

def choose_alpha_method(x,h,f_type='none'):
    if (f_type == 'quadratic'):

    else:
        return alpha_split(x,h)

def alpha_quadr(x,h): 
    return -np.dot((A.dot(x)+b),h)/np.dot(A.dot(h),h)

### STOP CONDITIONS
def stop1(x1,x2,k):
    plt.ylabel('|| x_new - x_old || ')
    d = norm(x2-x1, ord=2)
    plt.scatter(k, d)
    return d<=eps

def stop2(x1,x2,k):
    plt.ylabel('| f(x_new) - f(x_old) | ')
    d = abs(f(x2)-f(x1))
    plt.scatter(k, d)
    return d<=eps

def stop3(x,h,k):
    plt.ylabel('|| f\'(x_new) || ')
    d = norm(df(x,h), ord=2)
    plt.scatter(k, d)
    return d<=eps
